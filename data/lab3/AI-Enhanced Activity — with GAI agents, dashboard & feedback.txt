  :Ingest Tool Exports (axe/WAVE/HTML validator);

  end note

    note right

      - Keyboard path success

    end note

    note right
      Explains recurring UX/accessibility\npatterns, root-causes
    end note
    --> :Aggregate Manual & User Findings;
  end fork
}

:Merge Obtained Results;

partition "GAI Compliance Advisor" {
  :Compute Compliance Score\n& Priority Roadmap;
  note right
    - WCAG mapping + weights
    - Fix order suggestions
    - Est. remediation impact
  end note
  --> :Merge Obtained Results;
}

partition "Reporting & Feedback" {
  :Publish Interactive Dashboard;
  note right
    - Scores by site/page/criterion
    - Trends after re-tests
    - Dev checklist export
  end note
  if (Meets acceptance threshold?) then (Yes)
    :Finalize Accessibility Report;
    stop
  else (No)
    :Adjust thresholds / request re-tests;
    :Advisor triggers feedback loop;
    --> :Ingest Tool Exports (axe/WAVE/HTML validator);
  endif
}
